```javascript
function _selectSnipers(sniperCandidates, config, tierCuts) {
  if (!Array.isArray(sniperCandidates)) return [];
  
  var selected = [];
  var hqCount = 0;
  
  for (var i = 0; i < sniperCandidates.length; i++) {
    var s = sniperCandidates[i];
    if (!s) continue;
    
    // ═══════════════════════════════════════════════════════════
    // HQ PICKS: Always include if they passed the gate
    // ═══════════════════════════════════════════════════════════
    if (s.signalType === 'HIGH_QTR' || s.isHighQtr === true) {
      selected.push(s);
      hqCount++;
      continue; // Skip all other filters
    }
    
    // ═══════════════════════════════════════════════════════════
    // MARGIN PICKS: Check minimum margin
    // ═══════════════════════════════════════════════════════════
    if (s.signalType === 'MARGIN') {
      var marginMatch = String(s.pick || '').match(/[+-](\d+\.?\d*)/);
      if (!marginMatch) continue;
      
      var margin = parseFloat(marginMatch[1]);
      if (margin < config.sniperMinMargin) continue;
      
      selected.push(s);
      continue;
    }
    
    // ═══════════════════════════════════════════════════════════
    // OU PICKS: Check confidence/edge/EV filters
    // ═══════════════════════════════════════════════════════════
    if (config.includeOUSignals && 
        (s.signalType === 'OU' || s.signalType === 'OU_STAR' || 
         s.signalType === 'OU_HQ' || s.signalType === 'OU_DIR' || 
         s.signalType === 'OU_DIR_HQ')) {
      
      var conf = parseFloat(s.confidence) || 0;
      var ev = parseFloat(s.ev) || 0;
      var edge = s.edge;
      
      // Confidence OR EV must meet minimums
      var meetsConfOrEV = (conf >= config.ouMinConf) || (ev >= config.ouMinEv);
      if (!meetsConfOrEV) continue;
      
      // Edge gating (if configured)
      if (config.minEdgeScore > 0) {
        if (!isFinite(edge)) {
          // No edge value - require EV
          if (ev < config.ouMinEv) continue;
        } else if (edge === 0) {
          // Edge is 0 - require EV present
          if (!isFinite(ev) || ev < config.ouMinEv) continue;
        } else {
          // Edge has value - check minimum
          if (edge < config.minEdgeScore) continue;
        }
      }
      
      selected.push(s);
      continue;
    }
    
    // Unknown signal type - include it anyway
    selected.push(s);
  }
  
  Logger.log('[_selectSnipers] ' + sniperCandidates.length + ' candidates → ' + 
    selected.length + ' snipers (HQ: ' + hqCount + ') | ' +
    'HQ(minConf=' + config.hq_min_confidence + ', enabled=' + config.hqEnabled + ')');
  
  return selected;
}
```

## What This Does:

1. **HQ picks bypass ALL filters** - they go straight to the selected array
2. **Margin picks** only check the margin threshold
3. **OU picks** check confidence/EV/edge filters
4. **Logs HQ count** so you can verify they're being selected
