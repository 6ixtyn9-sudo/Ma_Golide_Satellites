## The Bug

The code loads quarter book lines from UpcomingClean:

```javascript
var q1Idx = ucMap['q1'];
var q2Idx = ucMap['q2'];
var q3Idx = ucMap['q3'];
var q4Idx = ucMap['q4'];
```

**BUT** this is only used to calculate the FT line fallback (sum of Q1+Q2+Q3+Q4). It's **never passed to the OU candidate builder**.

Later in the code (~line 450-550), when processing OU signals from `t2.ou`, it creates sniper candidates like this:

```javascript
var pickOU = _canonicalQOUPick_(oq, ouSig.direction, ouSig.line);
```

But `ouSig.line` comes from **OU_Log**, not from the Q1-Q4 columns in UpcomingClean!

## The Fix Needed

You need to:

1. **Build a quarter book line map** from UpcomingClean (Q1-Q4 columns)
2. **Pass it to `_loadTier2Signals`** so OU picks can match against actual book lines
3. **Use those book lines when creating OU candidates**

## Here's the fix to add:

Add this function right after `_loadUpcomingCleanFTLines()` (~line 280):

```javascript
// ═══════════════════════════════════════════════════════════════════════════
// QUARTER BOOK LINE LOOKUP FROM UPCOMINGCLEAN
// ═══════════════════════════════════════════════════════════════════════════
var _ucQuarterLineCache = null;

function _loadUpcomingCleanQuarterLines() {
  if (_ucQuarterLineCache !== null) return _ucQuarterLineCache;

  _ucQuarterLineCache = {};
  var ucSheet = ss.getSheetByName('UpcomingClean');
  if (!ucSheet) {
    log('UpcomingClean not found — cannot load quarter lines');
    return _ucQuarterLineCache;
  }

  var ucData = ucSheet.getDataRange().getValues();
  if (ucData.length < 2) return _ucQuarterLineCache;

  var ucMap = {};
  for (var i = 0; i < ucData[0].length; i++) {
    var h = String(ucData[0][i] || '').toLowerCase().trim().replace(/[\s_]+/g, '_');
    ucMap[h] = i;
  }

  var homeIdx = ucMap['home'];
  var awayIdx = ucMap['away'];
  var q1Idx = ucMap['q1'];
  var q2Idx = ucMap['q2'];
  var q3Idx = ucMap['q3'];
  var q4Idx = ucMap['q4'];

  for (var r = 1; r < ucData.length; r++) {
    var ucRow = ucData[r];
    var ucHome = String(ucRow[homeIdx] || '').trim().toLowerCase();
    var ucAway = String(ucRow[awayIdx] || '').trim().toLowerCase();
    if (!ucHome || !ucAway) continue;

    var key = ucHome + ' vs ' + ucAway;
    
    _ucQuarterLineCache[key] = {
      Q1: q1Idx !== undefined ? parseFloat(ucRow[q1Idx]) : null,
      Q2: q2Idx !== undefined ? parseFloat(ucRow[q2Idx]) : null,
      Q3: q3Idx !== undefined ? parseFloat(ucRow[q3Idx]) : null,
      Q4: q4Idx !== undefined ? parseFloat(ucRow[q4Idx]) : null
    };
  }

  log('Loaded quarter lines for ' + Object.keys(_ucQuarterLineCache).length + ' games');
  return _ucQuarterLineCache;
}

function _getQuarterLineFromUpcomingClean(homeTeam, awayTeam, quarter) {
  var cache = _loadUpcomingCleanQuarterLines();
  var key = (homeTeam + ' vs ' + awayTeam).toLowerCase();
  var lines = cache[key];
  if (!lines || !lines[quarter]) return null;
  var val = parseFloat(lines[quarter]);
  return (isFinite(val) && val > 0) ? val : null;
}
```

This will make the OU picks use the **actual book lines from UpcomingClean's Q1-Q4 columns** instead of just the predicted lines from OU_Log.
